function IsBridgeAddressSource(IPAddressSource)
{
if(("Static"==IPAddressSource)||("DHCP"==IPAddressSource))
{
return false;
}
return true;
}

function DHCPInfo(domain,name,ip,mac,remaintime,devtype, interfacetype,AddressSource)
{
var IsBridgeFlag = IsBridgeAddressSource(AddressSource);

if(IsBridgeFlag)
{
this.devtype    = "--";
this.ip = "--";
}
else
{
this.devtype    = devtype;
this.ip = ip;
}

this.domain= domain;
this.name = name;
this.mac        = mac;
this.remaintime= remaintime;
this.interfacetype = interfacetype;
this.AddressSource = AddressSource;
}
